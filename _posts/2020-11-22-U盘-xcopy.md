# U盘驱动器标签列表

```
import os,re

data = os.popen("wmic VOLUME GET Label").read() # 获取驱动器标签
_del = re.findall("Label[\W\S]*系统\W*",data)   # 系统驱动器标签
data = data.replace(_del[0],'')                 # 删除系统驱动器标签
data = data.replace(" ","")
data = re.findall(".*[^\n]",data)               # U盘驱动器标签列表

print(data)
```

# 叠加

```
from moviepy.editor import *

clip1 = VideoFileClip("test.mp4")
clip = VideoFileClip("test.mp4").subclip(5,10) # 5-10秒

clip = clip.volumex(0.5) # 音量的 0.8倍
clip1 = clip.volumex(2)  # 音量的 2倍

video = CompositeVideoClip([clip, clip1]) # 叠加索引 0 在最下

video.write_videofile("132.mp4", fps=60, remove_temp=True) # 保存
```

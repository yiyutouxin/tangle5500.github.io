# sympy

```
# 3*y = 18
# 2*x = 6

from sympy import Symbol, symbols, Eq, pprint, solve

x, y = symbols('x y') # 未知数

eq1 = Eq(3*y,18) # 3*y = 18
eq2 = Eq(2*x,6)  # 2*x = 6

pprint(eq1)      # 3*y = 18
pprint(eq2)      # 2*x = 6

sol = solve([eq1,eq2] ,[x,y])
print(sol) # {y: 6, x: 3}
```
